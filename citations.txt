"""
System-level REST routes: memory reporting and cache clearing.

Framework-agnostic — probes backends in priority order:
  1. torch.mps  (Apple Silicon)
  2. torch.cuda (NVIDIA)
  3. Falls back to 0 / no-op if no framework is available
"""
from __future__ import annotations

from typing import Any, Dict

from fastapi import APIRouter

router = APIRouter()


def _get_allocated_bytes() -> int:
    try:
        import torch
        if torch.backends.mps.is_available():
            return torch.mps.current_allocated_memory()
        if torch.cuda.is_available():
            return torch.cuda.memory_allocated()
    except Exception:
        pass
    return 0


def _get_reserved_bytes() -> int:
    try:
        import torch
        if torch.backends.mps.is_available():
            return torch.mps.current_allocated_memory()
        if torch.cuda.is_available():
            return torch.cuda.memory_reserved()
    except Exception:
        pass
    return 0


def _clear_cache() -> str:
    cleared = []
    try:
        import torch
        if torch.backends.mps.is_available():
            torch.mps.empty_cache()
            cleared.append("mps")
        if torch.cuda.is_available():
            torch.cuda.empty_cache()
            cleared.append("cuda")
    except Exception as exc:
        return f"error: {exc}"
    return f"cleared: {', '.join(cleared)}" if cleared else "no cache to clear"


@router.get("/system/memory")
async def get_memory() -> Dict[str, Any]:
    return {
        "allocatedBytes": _get_allocated_bytes(),
        "reservedBytes": _get_reserved_bytes(),
    }


@router.post("/system/clear-cache")
async def clear_cache() -> Dict[str, Any]:
    result = _clear_cache()
    allocated = _get_allocated_bytes()
    print(f"[system] clear-cache — {result}", flush=True)
    return {"ok": True, "result": result, "allocatedBytes": allocated}
```

  3. Falls back to 0 / no-op if no framework is available

    try:
        import torch
        if torch.backends.mps.is_available():
            return torch.mps.current_allocated_memory()
        if torch.cuda.is_available():
            return torch.cuda.memory_allocated()
    except Exception:
        pass
    return 0

    try:
        import torch
        if torch.backends.mps.is_available():
            return torch.mps.current_allocated_memory()
        if torch.cuda.is_available():
            return torch.cuda.memory_reserved()
    except Exception:
        pass
    return 0

            cleared.append("cuda")

"""
System-level REST routes: memory reporting and cache clearing.

Framework-agnostic — probes backends in priority order:
  1. torch.mps  (Apple Silicon)
  2. torch.cuda (NVIDIA)
  3. Falls back to 0 / no-op if no framework is available
"""
from __future__ import annotations

from typing import Any, Dict

from fastapi import APIRouter

router = APIRouter()


def _get_allocated_bytes() -> int:
    try:
        import torch
        if torch.backends.mps.is_available():
            return torch.mps.current_allocated_memory()
        if torch.cuda.is_available():
            return torch.cuda.memory_allocated()
    except Exception:
        pass
    return 0


def _get_reserved_bytes() -> int:
    try:
        import torch
        if torch.backends.mps.is_available():
            return torch.mps.current_allocated_memory()
        if torch.cuda.is_available():
            return torch.cuda.memory_reserved()
    except Exception:
        pass
    return 0


def _clear_cache() -> str:
    cleared = []
    try:
        import torch
        if torch.backends.mps.is_available():
            torch.mps.empty_cache()
            cleared.append("mps")
        if torch.cuda.is_available():
            torch.cuda.empty_cache()
            cleared.append("cuda")
    except Exception as exc:
        return f"error: {exc}"
    return f"cleared: {', '.join(cleared)}" if cleared else "no cache to clear"


@router.get("/system/memory")
async def get_memory() -> Dict[str, Any]:
    return {
        "allocatedBytes": _get_allocated_bytes(),
        "reservedBytes": _get_reserved_bytes(),
    }


@router.post("/system/clear-cache")
async def clear_cache() -> Dict[str, Any]:
    result = _clear_cache()
    allocated = _get_allocated_bytes()
    print(f"[system] clear-cache — {result}", flush=True)
    return {"ok": True, "result": result, "allocatedBytes": allocated}

"""
System-level REST routes: memory reporting and cache clearing.

Framework-agnostic — probes backends in priority order:
  1. torch.mps  (Apple Silicon)
  2. torch.cuda (NVIDIA)
  3. Falls back to 0 / no-op if no framework is available
"""
from __future__ import annotations

from typing import Any, Dict

from fastapi import APIRouter

router = APIRouter()


def _get_allocated_bytes() -> int:
    try:
        import torch
        if torch.backends.mps.is_available():
            return torch.mps.current_allocated_memory()
        if torch.cuda.is_available():
            return torch.cuda.memory_allocated()
    except Exception:
        pass
    return 0


def _get_reserved_bytes() -> int:
    try:
        import torch
        if torch.backends.mps.is_available():
            return torch.mps.current_allocated_memory()
        if torch.cuda.is_available():
            return torch.cuda.memory_reserved()
    except Exception:
        pass
    return 0


def _clear_cache() -> str:
    cleared = []
    try:
        import torch
        if torch.backends.mps.is_available():
            torch.mps.empty_cache()
            cleared.append("mps")
        if torch.cuda.is_available():
            torch.cuda.empty_cache()
            cleared.append("cuda")
    except Exception as exc:
        return f"error: {exc}"
    return f"cleared: {', '.join(cleared)}" if cleared else "no cache to clear"


@router.get("/system/memory")
async def get_memory() -> Dict[str, Any]:
    return {
        "allocatedBytes": _get_allocated_bytes(),
        "reservedBytes": _get_reserved_bytes(),
    }


@router.post("/system/clear-cache")
async def clear_cache() -> Dict[str, Any]:
    result = _clear_cache()
    allocated = _get_allocated_bytes()
    print(f"[system] clear-cache — {result}", flush=True)
    return {"ok": True, "result": result, "allocatedBytes": allocated}
