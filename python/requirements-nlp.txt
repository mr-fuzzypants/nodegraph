# Optional NLP dependencies for AnonymizerNode and SummarizerNode.
#
# Install:
#   pip install -r python/requirements-nlp.txt
#   python -m spacy download en_core_web_lg
#
# AnonymizerNode uses Microsoft Presidio to detect and anonymize PII
# before any text is sent to an LLM. SummarizerNode uses the OpenAI
# chat API (already covered by requirements-langchain.txt).
#
# Presidio ships two packages:
#   presidio-analyzer   — detects PII spans using spaCy + regex recognizers
#   presidio-anonymizer — applies operators (replace/redact/hash/mask) to spans
#
# spaCy model sizes:
#   en_core_web_sm  ~12 MB  — fast, lower recall on names
#   en_core_web_lg  ~750 MB — best accuracy (recommended for production)

presidio-analyzer>=2.2.354
presidio-anonymizer>=2.2.354
spacy>=3.7,<4.0
